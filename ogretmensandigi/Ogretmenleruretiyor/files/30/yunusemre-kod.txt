    <script>
        // Sound System
        const sounds = {
            tik: document.getElementById('snd-tik'),
            dong: document.getElementById('snd-dong'),
            cinlama: document.getElementById('snd-cinlama')
        };

        function playSound(name) {
            try {
                if(sounds[name]) {
                    sounds[name].currentTime = 0;
                    sounds[name].play().catch(e => console.log('Audio play error:', e));
                }
            } catch(e) {
                console.log("Ses oynatÄ±lamadÄ±", e);
            }
        }

        // State Management
        const state = {
            currentStage: 0,
            woodScore: 0,
            versesCompleted: 0
        };

        const config = {
            totalWoodTargets: 5,
            totalVerses: 2,
            woodAreaHeight: 500,
            woodAreaWidth: 0 // Will be set on init
        };

        // DOM Elements
        const progressBar = document.getElementById('progress-bar');
        const stages = [
            document.getElementById('stage-0'),
            document.getElementById('stage-1'),
            document.getElementById('stage-2'),
            document.getElementById('stage-3'),
            document.getElementById('stage-4')
        ];

        // --- Navigation Logic ---
        function nextStage() {
            playSound('cinlama'); // GeÃ§iÅŸ sesi

            // Hide current
            stages[state.currentStage].classList.add('hidden');
            stages[state.currentStage].classList.remove('flex'); // Just in case

            // Increment
            state.currentStage++;
            
            // Show next
            const next = stages[state.currentStage];
            next.classList.remove('hidden');
            
            // Progress Bar
            const progress = (state.currentStage / (stages.length - 1)) * 100;
            progressBar.style.width = `${progress}%`;

            // Stage Specific Init
            if (state.currentStage === 0) {
                 // Stop video if leaving stage 0
                 const vid = document.getElementById('main-video');
                 if(vid) vid.pause();
            }
            if (state.currentStage === 2) initWoodGame();
            if (state.currentStage === 4) fireConfetti();
            
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        // --- Stage 1: Choice Logic ---
        function handleChoice(choice) {
            const resultWheat = document.getElementById('wheat-result');
            const resultHimmet = document.getElementById('himmet-result');
            const nextBtn = document.getElementById('stage-1-next');

            if (choice === 'wheat') {
                playSound('dong');
                resultWheat.classList.remove('hidden');
                // Simulate Yunus realizing mistake
                setTimeout(() => {
                    nextBtn.classList.remove('hidden');
                }, 1500);
            } else {
                playSound('tik');
                resultHimmet.classList.remove('hidden');
                setTimeout(() => {
                    fireConfetti({ particleCount: 50, spread: 60, origin: { y: 0.7 } });
                    nextBtn.classList.remove('hidden');
                }, 800);
            }
        }

        // --- Stage 2: Wood Game Logic ---
        function initWoodGame() {
            const area = document.getElementById('wood-game-area');
            area.innerHTML = ''; // Clear
            
            // Get dimensions
            const rect = area.getBoundingClientRect();
            const w = rect.width;
            const h = rect.height || 400;

            // Add background basket
            const basket = document.createElement('div');
            basket.id = 'basket';
            basket.className = 'absolute bottom-4 right-4 w-24 h-24 bg-contain bg-no-repeat opacity-50 pointer-events-none z-0';
            basket.style.backgroundImage = "url('data:image/svg+xml;utf8,<svg xmlns=\\'http://www.w3.org/2000/svg\\' viewBox=\\'0 0 24 24\\' fill=\\'%23a16207\\'><path d=\\'M2 20h20v2H2v-2zm2-8h2v7H4v-7zm4 0h2v7H8v-7zm4 0h2v7h-2v-7zm4 0h2v7h-2v-7z\\'/></svg>')";
            area.appendChild(basket);

            const totalPieces = 15; // Mix of straight and crooked
            const straightCount = 5;

            // Generate Straight
            for(let i=0; i<straightCount; i++) createWood(area, true, w, h);
            // Generate Crooked
            for(let i=0; i<totalPieces - straightCount; i++) createWood(area, false, w, h);
        }

        function createWood(container, isStraight, w, h) {
            const el = document.createElement('div');
            el.className = `wood-piece ${isStraight ? 'wood-straight' : 'wood-crooked'}`;
            
            // Random Position & Rotation
            const angle = Math.random() * 360;
            const left = Math.random() * (w - 100);
            const top = Math.random() * (h - 100);

            if(isStraight) {
                el.style.width = '100px';
                el.style.height = '16px';
            }

            el.style.left = `${left}px`;
            el.style.top = `${top}px`;
            el.style.transform = `rotate(${angle}deg)`;

            // Click Handler
            el.onclick = function(e) {
                e.stopPropagation();
                if(isStraight) {
                    // Success logic
                    collectWood(el);
                } else {
                    // Fail logic
                    shakeElement(el);
                }
            };

            container.appendChild(el);
        }

        function collectWood(el) {
            playSound('tik'); // DoÄŸru ses

            // Animate flying to basket (simplified as fade out for now)
            el.style.transform = 'scale(0.1) rotate(0deg)';
            el.style.opacity = '0';
            setTimeout(() => el.remove(), 300);

            state.woodScore++;
            document.getElementById('score-counter').innerText = state.woodScore;

            if(state.woodScore >= config.totalWoodTargets) {
                setTimeout(() => {
                    document.getElementById('stage-2-success').classList.remove('hidden');
                    document.getElementById('stage-2-success').classList.add('flex');
                    fireConfetti();
                }, 500);
            }
        }

        function shakeElement(el) {
            playSound('dong'); // YanlÄ±ÅŸ ses

            // Visual feedback for wrong choice
            el.style.filter = 'sepia(1) hue-rotate(-50deg) saturate(3)'; // Turn red-ish
            let currentTransform = el.style.transform;
            el.style.transform = currentTransform + ' translateX(5px)';
            setTimeout(() => {
                el.style.transform = currentTransform;
                el.style.filter = '';
            }, 200);
        }

        // --- Stage 3: Verse Logic ---
        const answers = {
            0: 'kimseye',
            1: 'Yaradan\'dan'
        };

        function checkVerse(btn, index, selectedWord) {
            const card = document.querySelector(`.verse-card[data-index="${index}"]`);
            const slot = card.querySelector('.target-slot');
            
            if(selectedWord === answers[index]) {
                // Correct
                playSound('tik');
                btn.classList.add('correct-anim');
                slot.innerText = selectedWord;
                slot.classList.remove('text-teal-600');
                slot.classList.add('text-green-600');
                
                // Disable siblings
                const siblings = btn.parentElement.children;
                for(let sib of siblings) {
                    sib.disabled = true;
                    if(sib !== btn) sib.classList.add('opacity-50');
                }

                state.versesCompleted++;

                // Unlock next or finish
                if(index < config.totalVerses - 1) {
                    const nextCard = document.querySelector(`.verse-card[data-index="${index+1}"]`);
                    nextCard.classList.remove('opacity-50', 'pointer-events-none');
                } else {
                    document.getElementById('stage-3-complete').classList.remove('hidden');
                    fireConfetti();
                }

            } else {
                // Wrong
                playSound('dong');
                btn.classList.add('wrong-anim');
                setTimeout(() => btn.classList.remove('wrong-anim'), 400);
            }
        }
